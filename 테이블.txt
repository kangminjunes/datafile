


CREATE TABLE QUESTION_T (			// Q&A 테이블	
    QUESTION_NO    NUMBER             NOT NULL,	// 게시글 번호 
    TITLE        VARCHAR2(500 BYTE) NOT NULL,	//  제목
    CONTENTS     VARCHAR2(4000 BYTE),		// 내용
    USER_NO      NUMBER             NULL,	// 유저 번호(구분)
    CREATED_AT   VARCHAR2(30 BYTE),		// 작성일자
    TEXT_PW	 VARCHAR(16 BYTE),		// 게시글 비번 8자리(영어 숫자 가능)
    CONSTRAINT PK_UPLOAD PRIMARY KEY(QUESTION_NO),
    CONSTRAINT FK_USER_UPLOAD FOREIGN KEY(USER_NO) REFERENCES USER_T(USER_NO) ON DELETE SET NULL
);




CREATE TABLE BLOG_IMAGE_T (
    QUESTION_NO         NUMBER             NOT NULL,  //  테이블 번호
    IMAGE_PATH      VARCHAR2(100 BYTE),	 	      //  경로
    FILESYSTEM_NAME VARCHAR2(100 BYTE),		      // 파일명
    CONSTRAINT FK_BLOG_IMAGE FOREIGN KEY(BLOG_NO) REFERENCES BLOG_T(QUESTION_NO) ON DELETE CASCADE
);
